\begin{frame}{Previous literature}
    \metroset{block = fill}
    Differences in physician practice styles drive variation in healthcare utilization and expenditures (\cite{cutler_physician_2019}; \cite{badinski_geographic_2023}).

    \vspace{5mm}
    \begin{block}{But how do these practice styles develop?}
        \begin{itemize}
            \item \textbf{Formal training} seems to have some effect on treatment decisions. 
                \begin{itemize}
                    \item Physicians who attended lower-ranked medical schools tend to prescribe more opioids compared to those from higher-ranked medical schools (\cite{schnell_addressing_2018}). 
                    \item Patients randomly assigned to care teams from a higher-ranked residency program incur lower costs (\cite{doyle_returns_2010}). 
                \end{itemize}
            \item But \textbf{practice environment} seems to explain more in certain cases. 
                \begin{itemize}
                    \item Only 3\% of variation in c-section rates can be attributed to residency program (\cite{epstein_formation_2009}). 
                    \item Cardiologist practice patterns change immediately after a move (\cite{molitor_evolution_2018}). 
                \end{itemize}
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Economic model} \label{model}
    \metroset{block = fill}
    \vspace{2.5mm}
    Consider the physician agency model in \textcite{badinski_geographic_2023}.
        \begin{itemize}
            \item Patient $i$ is treated by physician $d$ at location $j$ at time $t$. 
            \item Patient $i$'s utility from treatment level $y$ is  
                \begin{equation*}
                    u_{idt}(y) = a_i y - \frac{1}{2}(y-h_{idt})^2,
                \end{equation*}
            where $a_i$ is patient-specific and $h_{idt}$ is patient $i$'s optimal level of care. 
        \end{itemize}
\end{frame}

\begin{frame}{Economic model}
        \begin{itemize}
            \item The physician's cost of treatment is 
                \begin{equation*}
                    c_{idt} = (c_{j(it)} + g_{idt})y, 
                \end{equation*}
            where $c_{j}$ includes institutional factors and $g_{idt}$ is marginal cost.
            \item Physician $d$ chooses treatment $y$ that maximizes their utility.  
                \begin{equation*}
                    y_{idt} = \arg\max_y \left\{u_{idt}(y) - c_{idt}(y) - b_d \{\text{cost to patient}\}_{it}\right\}, 
                \end{equation*}
            where $b_d$ is physician-specific. 
        \end{itemize}
